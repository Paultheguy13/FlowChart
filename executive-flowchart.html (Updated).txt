    <script>
        let currentPath = "Start";
        
        function updatePath(step) {
            currentPath += " ‚Üí " + step;
            document.getElementById('currentPath').textContent = currentPath;
        }
        
        function hideAll() {
            const nodes = document.querySelectorAll('.node');
            nodes.forEach(node => {
                node.classList.add('hidden');
                node.classList.remove('fade-in');
            });
        }
        
        function showNode(nodeId) {
            hideAll();
            const node = document.getElementById(nodeId);
            node.classList.remove('hidden');
            node.classList.add('fade-in');
            // Removed scroll behavior to prevent iframe scrolling issues
        }
        
        function showQuestion1() {
            updatePath("Strategic Deal?");
            showNode('q1');
        }
        
        function showQuestion2() {
            updatePath("E2E Relationships?");
            showNode('q2');
        }
        
        function showQuestion3() {
            updatePath("High-level Roadblock?");
            showNode('q3');
        }
        
        function showQuestion4() {
            updatePath("Executive Commitment?");
            showNode('q4');
        }
        
        function showQuestion5() {
            updatePath("Critical Message?");
            showNode('q5');
        }
        
        function showAnswer(answerId) {
            let answerText = "";
            switch(answerId) {
                case 'q1-yes': answerText = "Yes ‚Üí Involve Executive"; break;
                case 'q2-yes': answerText = "Yes ‚Üí Involve Executive"; break;
                case 'q3-yes': answerText = "Yes ‚Üí Involve Executive"; break;
                case 'q4-yes': answerText = "Yes ‚Üí Involve Executive"; break;
                case 'q5-yes': answerText = "Yes ‚Üí Involve Executive"; break;
                case 'final-no': answerText = "No ‚Üí Don't Involve Executive"; break;
            }
            updatePath(answerText);
            showNode(answerId);
        }
        
        function restart() {
            currentPath = "Start";
            document.getElementById('currentPath').textContent = currentPath + " ‚Üí What's your objective?";
            showNode('start');
        }
